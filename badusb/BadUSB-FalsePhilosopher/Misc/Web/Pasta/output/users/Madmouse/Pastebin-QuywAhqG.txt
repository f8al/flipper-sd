[bits 32]
section .text
global start

org 0x08048000
ehdr:                                      ; Elf32_Ehdr
	db 0x7F,"ELF" ;, 1, 1, 1, 0         ;   e_ident
;                   ^ 4 more bytes for code caving
;	times 8 db 0
; We can replace ^ this with the following for a code cave inside the elf header itself
start:
	xor eax, eax
	xor ebx, ebx
	inc eax
	int 0x80
end:
	times 12-(end-start) db 'A'	; we need bytes for padding, the code cave here is 12 bytes long
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
	dw 2                               ;   e_type
	dw 3                               ;   e_machine

	dd 0                               ;   e_flags

	dw 1                               ;   e_phnum
	dw 0                               ;   e_shentsize
	dw 0                               ;   e_shnum
	dw 0                               ;   e_shstrndx
	ehdrsize equ $ - ehdr

phdr:                                      ; Elf32_Phdr
	dd 1                               ;   p_type
	dd 0                               ;   p_offset
	dd $$                              ;   p_vaddr
	dd $$                              ;   p_paddr
	dd filesize                        ;   p_filesz
	dd filesize                        ;   p_memsz
	dd 7                               ;   p_flags
	dd 0x1000                          ;   p_align
	phdrsize equ $ - phdr
